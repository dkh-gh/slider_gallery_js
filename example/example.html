Slider Gallery JS v0.1.1

Slider Gallery JS is a simple use horisontal slider.
Watch examples here,
and documentation on github.

#slider1

default settings

slider1 = new SliderGallery(
  document.querySelector('#slider1')
);

#slider2

more space, no rotation, weak scaling, faster

slider2 = new SliderGallery(
  document.querySelector('#slider2')
);
slider2.configure({
  'slide_poses_count': 3,
  'slide_max_rot_angle': 0,
  'slide_min_scale': .6,
});

#slider3

page turning

slider3 = new SliderGallery(
  document.querySelector('#slider3')
);
slider2.configure({
  'slide_poses_count': 3,
  'slide_max_rot_angle': 110,
  'slide_min_scale': .5,
  'reverse_rotation': true,
  'perspective': 750,
  'transition': .5,
});
